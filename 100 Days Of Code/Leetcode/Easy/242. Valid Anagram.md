```python
class Solution:

    def isAnagram(self, s: str, t: str) -> bool:

        if len(s) != len(t):

            return False

        ana_one = {}

        ana_two = {}

        for val in s:

            # print(val)

            if val in ana_one:

                ana_one[val] += 1

            else:

                ana_one[val] = 1

  

        for val in t:

            if val in ana_two:

                ana_two[val] += 1

            else:

                ana_two[val] = 1

        for key, value in ana_one.items():

            if key in ana_two:

                val = ana_two[key]

                if value != val:

                    return False

            else:

                return False

        return True
```
